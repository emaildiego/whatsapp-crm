const STORE='wa-crm-v2';

const TAG_COLORS={key:'#8b5cf6',work:'#3b82f6',friend:'#ec4899',family:'#f59e0b',client:'#06b6d4',mentor:'#10b981'};

let contacts=JSON.parse(localStorage.getItem(STORE)||'null')||[

  {id:1,name:'Sarah Johnson',phone:'+14155552671',tags:['work','key'],lastNote:'Just got promoted to VP of Marketing',lastContact:'2026-02-01',reminderDays:30},

  {id:2,name:'Mike Chen',phone:'+14155559823',tags:['friend','key'],lastNote:'Training for his first marathon in April',lastContact:'2026-01-15',reminderDays:30},

];

let expanded=null,messages={},currentView='dashboard',editingNote=null,showForm=false,search='';

let apiKey=localStorage.getItem('wa-crm-apikey')||'';

let notifPermission=localStorage.getItem('wa-notif')||'default';

let formTags=[];


function save(){localStorage.setItem(STORE,JSON.stringify(contacts));document.getElementById('contact-count').textContent=contacts.length+' contacts';}

function daysAgo(d){if(!d)return 999;return Math.floor((Date.now()-new Date(d).getTime())/86400000);}

function dotColor(days,limit){if(days>=limit)return'#ef4444';if(days>=limit*0.75)return'#f59e0b';return'#22c55e';}

function tagChip(t){const c=TAG_COLORS[t]||'#475569';return`<span class="tag" style="background:${c}33;color:${c}">${t==='key'?'⭐':t}</span>`;}

function tagBtn(t,active){const c=TAG_COLORS[t]||'#475569';return`<button class="tag-btn" onclick="toggleTag(${expanded},'${t}')" style="${active?`background:${c}33;color:${c};border-color:${c}55`:''}">${t==='key'?'⭐ key':t}</button>`;}


async function generateMessage(id){

  if(!apiKey){alert('Please add your Anthropic API key in ⚙ Settings first.');return;}

  const c=contacts.find(x=>x.id===id);

  messages[id]='...';render();

  try{

    const res=await fetch('https://api.anthropic.com/v1/messages',{

      method:'POST',

      headers:{'Content-Type':'application/json','x-api-key':apiKey,'anthropic-version':'2023-06-01','anthropic-dangerous-direct-browser-access':'true'},

      body:JSON.stringify({model:'claude-sonnet-4-20250514',max_tokens:1000,messages:[{role:'user',content:`Write a short warm WhatsApp message to reach out to ${c.name}. Last note: "${c.lastNote||'No notes'}". Days since last contact: ${daysAgo(c.lastContact)}. Natural, genuine, 2-4 sentences, casual sign-off, no subject line.`}]})

    });

    const d=await res.json();

    messages[id]=d.content?.[0]?.text||'Could not generate.';

  }catch{messages[id]='Error. Check your API key in Settings.';}

  render();

}


function toggleExpand(id){expanded=expanded===id?null:id;editingNote=null;render();}

function toggleTag(id,t){const c=contacts.find(x=>x.id===id);c.tags=c.tags.includes(t)?c.tags.filter(x=>x!==t):[...c.tags,t];save();render();}

function markContacted(id){contacts.find(x=>x.id===id).lastContact=new Date().toISOString().split('T')[0];save();render();}

function updateFrequency(id,days){contacts.find(x=>x.id===id).reminderDays=parseInt(days);save();render();}

function deleteContact(id){if(!confirm('Delete this contact?'))return;contacts=contacts.filter(x=>x.id!==id);if(expanded===id)expanded=null;save();render();}

function saveNote(id){const ta=document.getElementById('note-ta');if(ta)contacts.find(x=>x.id===id).lastNote=ta.value;editingNote=null;save();render();}

function sendWA(id){const c=contacts.find(x=>x.id===id);const ta=document.getElementById(`msg-${id}`);const msg=ta?ta.value:(messages[id]||'');window.open(`https://wa.me/${c.phone.replace(/\D/g,'')}?text=${encodeURIComponent(msg)}`,'_blank');}


async function importContacts(){

  if(!('contacts' in navigator&&'ContactsManager' in window)){alert('Contact picker is only available in Safari on iOS 14.5+.\n\nMake sure you\'ve installed this app via "Add to Home Screen" in Safari.');return;}

  try{

    const sel=await navigator.contacts.select(['name','tel'],{multiple:true});

    let added=0;

    sel.forEach(c=>{

      const name=c.name?.[0]||'Unknown';

      const phone=c.tel?.[0]||'';

      if(!phone)return;

      if(!contacts.some(x=>x.phone===phone||x.name===name)){

        contacts.push({id:Date.now()+Math.random(),name,phone,tags:[],lastNote:'',lastContact:null,reminderDays:30});

        added++;

      }

    });

    save();alert(`Imported ${added} new contact${added!==1?'s':''}.`);render();

  }catch(e){if(e.name!=='AbortError')alert('Could not access contacts: '+e.message);}

}


function toggleFormTag(t){formTags=formTags.includes(t)?formTags.filter(x=>x!==t):[...formTags,t];renderFormTags();}

function renderFormTags(){

  const el=document.getElementById('form-tags');if(!el)return;

  el.innerHTML=['key','work','friend','family','client','mentor'].map(t=>{

    const a=formTags.includes(t);const c=TAG_COLORS[t]||'#475569';

    return`<button class="tag-btn" onclick="toggleFormTag('${t}')" style="${a?`background:${c}33;color:${c};border-color:${c}55`:''}">${t==='key'?'⭐ key':t}</button>`;

  }).join('');

}

function submitForm(){

  const name=document.getElementById('f-name').value.trim();

  const phone=document.getElementById('f-phone').value.trim();

  const note=document.getElementById('f-note').value.trim();

  const days=parseInt(document.getElementById('f-days').value)||30;

  if(!name||!phone){alert('Name and phone are required.');return;}

  contacts.push({id:Date.now(),name,phone,tags:[...formTags],lastNote:note,lastContact:null,reminderDays:days});

  formTags=[];showForm=false;save();render();

}


async function requestNotifications(){

  if(!('Notification' in window)){alert('Notifications require Safari on iOS 16.4+ with the app installed via Add to Home Screen.');return;}

  const result=await Notification.requestPermission();

  notifPermission=result;localStorage.setItem('wa-notif',result);

  if(result==='granted'){scheduleCheck();alert('Notifications enabled! You\'ll be reminded for overdue contacts each time you open the app.');}

  else alert('Permission denied. Enable in iPhone Settings → Apps → Safari → Notifications.');

  render();

}

function scheduleCheck(){checkReminders();localStorage.setItem('wa-last-check',Date.now());}

function checkReminders(){

  if(!navigator.serviceWorker?.controller)return;

  navigator.serviceWorker.controller.postMessage({type:'CHECK_REMINDERS',contacts});

}

function checkOnOpen(){

  const last=parseInt(localStorage.getItem('wa-last-check')||'0');

  if(notifPermission==='granted'&&Date.now()-last>1000*60*60*12)scheduleCheck();

}

function setView(v){

  document.querySelectorAll('.nav-btn').forEach((b,i)=>{b.classList.toggle('active',['dashboard','contacts','settings'][i]===v);});

  render();

}

  if(currentView==='dashboard'){

    if(!overdue.length&&!upcoming.length)html+=`<div class="empty">🎉<br><br>All caught up!</div>`;

    if(overdue.length)html+=`<div class="section-header"><div class="dot" style="background:#ef4444"></div><span style="color:#ef4444">OVERDUE</span></div>${overdue.map(contactRowHTML).join('')}`;

    if(upcoming.length)html+=`<div class="section-header" style="margin-top:12px"><div class="dot" style="background:#f59e0b"></div><span style="color:#f59e0b">DUE SOON</span></div>${upcoming.map(contactRowHTML).join('')}`;

    body.innerHTML=html;return;

  }


  let html=`<div class="search-row">

    <input class="search" placeholder="Search…" value="${search}" oninput="search=this.value;render()"/>

    <button class="btn-import" onclick="importContacts()">📋 Import</button>

    <button class="btn-add" onclick="showForm=!showForm;render()">+</button>

  </div>`;

  if(showForm){

    html+=`<div class="form-card">

      <div class="form-title">NEW CONTACT</div>

      ${[['Name','f-name','text'],['Phone (with country code)','f-phone','tel'],['Last interaction note','f-note','text'],['Remind every (days)','f-days','number']].map(([l,id,t])=>`

        <div class="form-field"><div class="form-label">${l}</div>

        <input class="form-input" id="${id}" type="${t}" ${id==='f-days'?'value="30"':''}></div>`).join('')}

      <div class="form-field"><div class="form-label">TAGS</div><div class="tags-row" id="form-tags"></div></div>

      <div class="btn-row">

        <button class="btn btn-save" onclick="submitForm()" style="flex:1">Add Contact</button>

        <button class="btn btn-cancel" onclick="showForm=false;render()">Cancel</button>

      </div>

    </div>`;

  }

  html+=filtered.length?filtered.map(contactRowHTML).join(''):`<div class="empty">No contacts found</div>`;

  body.innerHTML=html;

  if(showForm)renderFormTags();

}
